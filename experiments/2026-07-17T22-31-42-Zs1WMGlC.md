import torch
import torch.nn as nn
import torch.nn.functional as F


class Attention(nn.Module):
    """Multi-head causal self-attention using PyTorch SDPA."""

    def __init__(self, cfg):
        super().__init__()

        if cfg.n_embd % cfg.n_head != 0:
            raise ValueError("n_embd must be divisible by n_head")

        self.n_embd = cfg.n_embd
        self.n_head = cfg.n_head
        self.head_dim = cfg.n_embd // cfg.n_head

        self.attn_dropout_p = getattr(
            cfg, "attn_dropout", cfg.dropout
        )
        self.resid_dropout_p = getattr(
            cfg, "resid_dropout", cfg.dropout
        )

        bias = getattr(cfg, "bias", False)

        # One fused projection is generally preferable to three separate ones.
        self.qkv = nn.Linear(
            self.n_embd,
            3 * self.n_embd,
            bias=bias,
        )

        self.proj = nn.Linear(
            self.n_embd,
            self.n_embd,
            bias=bias,
        )

        self.resid_dropout = nn.Dropout(self.resid_dropout_p)

        # Optional training-stability modification.
        self.use_qk_norm = getattr(cfg, "qk_norm", False)

        if self.use_qk_norm:
            self.q_norm = nn.RMSNorm(self.head_dim)
            self.k_norm = nn.RMSNorm(self.head_dim)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        B, T, C = x.shape

        if C != self.n_embd:
            raise ValueError(
                f"Expected embedding dimension {self.n_embd}, got {C}"
            )

        # (B, T, C) -> (B, T, 3, H, D)
        qkv = self.qkv(x).reshape(
            B,
            T,
            3,
            self.n_head,
            self.head_dim,
        )

        q, k, v = qkv.unbind(dim=2)

        # (B, T, H, D) -> (B, H, T, D)
        q = q.transpose(1, 2)
        k = k.transpose(1, 2)
        v = v.transpose(1, 2)

        if self.use_qk_norm:
            q = self.q_norm(q)
            k = self.k_norm(k)

        y = F.scaled_dot_product_attention(
            q,
            k,
            v,
            dropout_p=(
                self.attn_dropout_p if self.training else 0.0
            ),
            is_causal=True,
        )

        # (B, H, T, D) -> (B, T, C)
        y = y.transpose(1, 2).reshape(B, T, C)

        return self.resid_dropout(self.proj(y))
